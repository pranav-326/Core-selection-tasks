Read: GET Requests

Learning objectives

  • Write GET requests using `fetch`
  • Parse JSON and handle non-2xx responses
  • Use `fetchWithRetry` for idempotent reads

Example: fetchPosts()

async function fetchPosts() {
  // fetch example
  const res = await fetch(`${API_BASE}/posts`);
  if (!res.ok) throw new Error('Failed to fetch posts: ' + res.status);
  const data = await res.json();
  return data;
}

// XHR fallback
function fetchPostsXHR(cb) {
  const xhr = new XMLHttpRequest();
  xhr.open('GET', `${API_BASE}/posts`);
  xhr.onload = () => cb(null, JSON.parse(xhr.responseText));
  xhr.onerror = () => cb(new Error('Network error'));
  xhr.send();
}

PRO TIP: Use `fetchWithRetry` for unstable networks - see `js/lib/retry.js`
